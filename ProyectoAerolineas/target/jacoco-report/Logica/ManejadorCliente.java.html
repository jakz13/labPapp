package logica;

import DataTypes.DtCliente;
import jakarta.persistence.EntityManager;
import jakarta.persistence.EntityTransaction;
import jakarta.persistence.PersistenceException;
import jakarta.persistence.TypedQuery;
import java.time.LocalDate;
import java.util.HashMap;
import java.util.List;
import java.util.Map;
import java.util.ArrayList;
import java.util.Objects;

/**
 * Manejador singleton de clientes. Mantiene clientes en memoria y coordina
 * operaciones de persistencia y consulta.
 */
public final class ManejadorCliente {

    /** Mapa de clientes en memoria indexado por nickname. */
    private final Map<String, Cliente> clientes;
    private static volatile ManejadorCliente instancia = null;

    private ManejadorCliente() {
        clientes = new HashMap<>();
    }

    /**
     * Obtiene la instancia singleton del manejador de clientes.
     * @return instancia única de ManejadorCliente
     */
    public static ManejadorCliente getInstance() {
        if (instancia == null) {
            synchronized (ManejadorCliente.class) {
                if (instancia == null) {
                    instancia = new ManejadorCliente();
                }
            }
        }
        return instancia;
    }

    // =================== CRUD BD ===================
    /**
     * Carga todos los clientes desde la base de datos al mapa en memoria.
     * @param entManager EntityManager usado para la consulta
     */
    public void cargarClientesDesdeBD(EntityManager entManager) {
        TypedQuery<Cliente> query = entManager.createQuery("SELECT c FROM Cliente c", Cliente.class);
        List<Cliente> clientesPersistidos = query.getResultList();
        for (Cliente c : clientesPersistidos) {
            clientes.put(c.getNickname(), c);
        }
    }

    /**
     * Agrega un cliente nuevo validando unicidad y persistiendo en BD.
     * @param cliente objeto Cliente a agregar
     * @param entManager EntityManager utilizado para persistencia
     */
    public void agregarCliente(Cliente cliente, EntityManager entManager) {
        if (cliente == null) {
            throw new IllegalArgumentException("El cliente no puede ser null");
        }
        if (this.obtenerCliente(cliente.getNickname()) == null &&
                this.obtenerClientePorDocumento(cliente.getNumeroDocumento()) == null &&
                this.obtenerClientePorEmail(cliente.getEmail()) == null) {

            EntityTransaction entTransaction = entManager.getTransaction();
            try {
                entTransaction.begin();
                entManager.persist(cliente);
                clientes.put(cliente.getNickname(), cliente);
                entTransaction.commit();
            } catch (PersistenceException e) {
                if (entTransaction.isActive()) {
                    entTransaction.rollback();
                }
                clientes.remove(cliente.getNickname());
                throw new IllegalStateException("Error al persistir el cliente: " + e.getMessage(), e);
            }

        } else {
            throw new IllegalArgumentException("Ya existe un cliente con el mismo nickname, documento o email.");
        }
    }

    /**
     * Modifica datos parciales de un cliente y persiste los cambios.
     * @param clienteTemporal DTO con los datos a modificar
     * @param entManager EntityManager para la operación
     */
    public void modificarDatosCliente(DtCliente clienteTemporal, EntityManager entManager) {
        if (clienteTemporal == null) {
            throw new IllegalArgumentException("Los datos del cliente no pueden ser null");
        }
        Cliente clienteOriginal = obtenerClienteReal(clienteTemporal.getNickname());
        if (clienteOriginal == null) throw new IllegalArgumentException("Cliente no encontrado");

        if (clienteTemporal.getNombre() != null) clienteOriginal.setNombre(clienteTemporal.getNombre());
        if (clienteTemporal.getApellido() != null) clienteOriginal.setApellido(clienteTemporal.getApellido());
        if (clienteTemporal.getNacionalidad() != null) clienteOriginal.setNacionalidad(clienteTemporal.getNacionalidad());
        if (clienteTemporal.getTipoDocumento() != null) clienteOriginal.setTipoDocumento(clienteTemporal.getTipoDocumento());
        if (clienteTemporal.getNumeroDocumento() != null) clienteOriginal.setNumeroDocumento(clienteTemporal.getNumeroDocumento());
        if (clienteTemporal.getFechaNacimiento() != null) clienteOriginal.setFechaNacimiento(clienteTemporal.getFechaNacimiento());

        EntityTransaction entTransaction = entManager.getTransaction();
        try {
            entTransaction.begin();
            entManager.merge(clienteOriginal);
            entTransaction.commit();
        } catch (PersistenceException  e) {
            if (entTransaction.isActive()) {
                entTransaction.rollback();
            }
            throw new IllegalStateException("Error al modificar el cliente: " + e.getMessage(), e);
        }
    }

    // =================== Consultas en memoria ===================
    /**
     * Obtiene el DTO de un cliente por nickname.
     * @param nickname identificador del cliente
     * @return DtCliente o null si no existe
     */
    public DtCliente obtenerCliente(String nickname) {
        Cliente cliente = clientes.get(nickname);
        return (cliente != null) ? cliente.getDtCliente() : null;
    }

    /**
     * Obtiene el objeto Cliente real (entidad) por nickname.
     */
    public Cliente obtenerClienteReal(String nickname) {
        Cliente cliente = clientes.get(nickname);
        return cliente;
    }

    /**
     * Busca un cliente por número de documento.
     */
    public Cliente obtenerClientePorDocumento(String numeroDocumento) {
        return clientes.values().stream()
                .filter(c -> Objects.equals(c.getNumeroDocumento(), numeroDocumento))
                .findFirst().orElse(null);
    }

    /**
     * Busca un cliente por email.
     */
    public Cliente obtenerClientePorEmail(String email) {
        return clientes.values().stream()
                .filter(c -> Objects.equals(c.getEmail(), email))
                .findFirst().orElse(null);
    }

    // Mtodo para verificar login
    /**
     * Verifica las credenciales de un cliente por email y password.
     */
    public boolean verificarLogin(String email, String password) {
        Cliente cliente = obtenerClientePorEmail(email);
        if (cliente != null) {
            return cliente.verificarPassword(password);
        }
        return false;
    }

    /**
     * Actualiza la contrase1a de un cliente identificado por email.
     */
    public void actualizarPassword(String email, String nuevaPassword, EntityManager entManager) {
        Cliente cliente = obtenerClientePorEmail(email);
        if (cliente != null) {
            cliente.setPassword(nuevaPassword);
            EntityTransaction entTransaction = entManager.getTransaction();
            try {
                entTransaction.begin();
                entManager.merge(cliente);
                entTransaction.commit();
                //System.out.println("Contraseña actualizada para cliente: " + email);
            } catch (PersistenceException  e) {
                if (entTransaction.isActive()) entTransaction.rollback();
                throw new IllegalStateException("Error actualizando contraseña: " + e.getMessage(), e);
            }
        } else {
            throw new IllegalArgumentException("Cliente no encontrado: " + email);
        }
    }

    /**
     * Modifica todos los datos del cliente y persiste los cambios.
     */
    public void modificarDatosClienteCompleto(String nickname, String nombre, String apellido,
                                              String nacionalidad, LocalDate fechaNacimiento, TipoDoc tipoDocumento,
                                              String numeroDocumento, String password, String imagenUrl, EntityManager entManager) {
        Cliente cliente = obtenerClienteReal(nickname);
        if (cliente == null) {
            throw new IllegalArgumentException("Cliente no encontrado: " + nickname);
        }

        if (nombre != null) cliente.setNombre(nombre.trim());
        if (apellido != null) cliente.setApellido(apellido.trim());
        if (nacionalidad != null) cliente.setNacionalidad(nacionalidad.trim());
        if (fechaNacimiento != null) cliente.setFechaNacimiento(fechaNacimiento);
        if (tipoDocumento != null) cliente.setTipoDocumento(tipoDocumento);
        if (numeroDocumento != null) cliente.setNumeroDocumento(numeroDocumento.trim());
        if (password != null && !password.trim().isEmpty()) cliente.setPassword(password.trim());
        if (imagenUrl != null) cliente.setImagenUrl(imagenUrl.trim().isEmpty() ? null : imagenUrl.trim());

        EntityTransaction entTransaction = entManager.getTransaction();
        try {
            entTransaction.begin();
            entManager.merge(cliente);
            entTransaction.commit();
            //System.out.println("Todos los datos actualizados para cliente: " + nickname);
        } catch (PersistenceException e) {
            if (entTransaction.isActive()) entTransaction.rollback();
            throw new IllegalStateException("Error modificando datos completos del cliente: " + e.getMessage(), e);
        }
    }

    /**
     * Devuelve la lista de clientes en forma de DTO.
     */
    public List<DtCliente> getClientes() {
        List<DtCliente> lista = new ArrayList<>();
        for (Cliente c : clientes.values()) {
            lista.add(c.getDtCliente());
        }
        return lista;
    }

    /**
     * Actualiza la URL de la imagen de un cliente y persiste el cambio.
     */
    public void actualizarImagenCliente(String nickname, String imagenUrl, EntityManager entManager) {
        Cliente cliente = clientes.get(nickname);
        if (cliente != null) {
            cliente.setImagenUrl(imagenUrl);
            EntityTransaction entTransaction = entManager.getTransaction();
            try {
                entTransaction.begin();
                entManager.merge(cliente);
                entTransaction.commit();
                //System.out.println("Imagen actualizada para cliente: " + nickname);
            } catch (PersistenceException e) {
                if (entTransaction.isActive()) entTransaction.rollback();
                throw new IllegalStateException("Error actualizando imagen del cliente: " + e.getMessage(), e);
            }
        } else {
            throw new IllegalArgumentException("Cliente no encontrado: " + nickname);
        }
    }

    /**
     * Agrega una reserva a la entidad Cliente y la persiste.
     */
    public void agregarReserva(Reserva reserva, String nicknameCliente, Long idReserva, EntityManager entManager) {
        DtCliente cliente = obtenerCliente(nicknameCliente);
        Cliente clienteObj = clientes.get(nicknameCliente);
        if (cliente == null || clienteObj == null) {
            throw new IllegalArgumentException("Cliente no encontrado");
        }

        try {
            clienteObj.agregarReserva(reserva);

            entManager.merge(clienteObj);

        } catch (PersistenceException e) {
            throw new IllegalStateException("Error al agregar la reserva: " + e.getMessage(), e);
        }
    }
}
